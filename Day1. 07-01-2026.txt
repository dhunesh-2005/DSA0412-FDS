## #4 total sales for the year and determine the percentage increase in sales
import numpy as np
import pandas as pd

df = pd.read_excel("/content/Sales.xlsx")

df['Month_sales'] = pd.to_datetime(df['Month_sales'])
df['Quarter'] = df['Month_sales'].dt.quarter
print(df['Quarter'])
sales_data = df.groupby('Quarter')['Total'].sum().values
print(sales_data)
total_sales = np.sum(sales_data)
percentage_increase = ((sales_data[3] - sales_data[0]) / sales_data[0]) * 100

print("Total Sales for the Year:", total_sales)
print("Percentage Increase from Q1 to Q4:", percentage_increase)


Output: 
Total Sales for the Year: 5019265.2299999995
Percentage Increase from Q1 to Q4: 4.224756773694266


#5 calculate the average fuel efficiency and determine the percentage improvement
import numpy as np

fuel_efficiency = np.array([22, 25, 28, 30, 35])

average_efficiency = np.mean(fuel_efficiency)
print(average_efficiency)
model_A = fuel_efficiency[1]   # 25 mpg
model_B = fuel_efficiency[4]   # 35 mpg

percentage_improvement = ((model_B - model_A) / model_A) * 100

print("Average Fuel Efficiency:", average_efficiency, "mpg")
print("Percentage Improvement:", percentage_improvement, "%")


Output:

Average Fuel Efficiency: 21.46
Percentage Improvement: 4.0
